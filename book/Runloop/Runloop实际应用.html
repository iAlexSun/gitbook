如何使用 Runloop 实现一个常驻线程？这种线程一般有什么作用？

  • 使用 NSMachPort port 来实现线程保活持续,类似于AFN的常驻线程机制.
  • 下载文件

在实际需求中有需要常驻线程的情况，在文章页面推荐流需要去发曝光信息，因为需要实时监测页面上tableView会有多少个数据，这时需要在子线程遍历每个cell的数据是否被曝光过情况。如果不使用常驻线程可能出现的情况就是会创建大量线程那么就用到了线程保活。

//初始化子线程
@property (nonatomic, strong) NSThread *thread;

self.thread = [[NSThread alloc] initWithTarget:self selector:@selector(subThreadEnter) object:nil];
[self.thread setName:@"track"];
[self.thread start];

//保活子线程
- (void)subThreadEnter {
    @autoreleasepool {
        NSRunLoop *runLoop = [NSRunLoop currentRunLoop];
        //如果注释了下面这一行，子线程中的任务并不能正常执行
        [runLoop addPort:[NSMachPort port] forMode:NSRunLoopCommonModes];
        NSLog(@"启动RunLoop前--%@",runLoop.currentMode);
        [runLoop run];
    }
}

//在指定线程中执行任务
- (void)trackShowWithIndexPaths:(NSArray<NSIndexPath *> *)indexPaths {
    self.indexPaths = indexPaths;
    [self performSelector:@selector(run) onThread:self.thread withObject:nil waitUntilDone:NO];
}

//需要执行的任务
- (void)run {
    NSLock *lock = [[NSLock alloc] init];
    [self.indexPaths enumerateObjectsUsingBlock:^(NSIndexPath * _Nonnull indexPath, NSUInteger idx, BOOL * _Nonnull stop) {
        [lock lock];
        SHMUBSTrackModel *trackModel = [self trackModelWithIndexPath:indexPath];
        if (!trackModel.isExposed && trackModel) {
            [[SHMUBSManager sharedInstance] EVStartWithTrackModel:trackModel];
            [self markExposedItemModelAtIndexPath:indexPath];
        }
        [lock unlock];
    }];
}
